DROP TABLE IF EXISTS wp_tsa_skater_scoringpergame;

CREATE TABLE wp_tsa_skater_scoringpergame (
    season VARCHAR(7),
    seasonId INT,
    gameTypeId INT,

    gameDate DATE,
    gameId BIGINT,

    skaterFullName VARCHAR(150),
    playerId BIGINT,

    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),

    shootsCatches VARCHAR(10),
    positionCode VARCHAR(10),

    gamesPlayed INT,

    goals INT,
    assists INT,
    totalPrimaryAssists INT,
    totalSecondaryAssists INT,
    points INT,
    shots INT,
    penaltyMinutes INT,
    hits INT,
    blockedShots INT,
    timeOnIce INT,

    goalsPerGame DECIMAL(10,4),
    assistsPerGame DECIMAL(10,4),
    primaryAssistsPerGame DECIMAL(10,4),
    secondaryAssistsPerGame DECIMAL(10,4),
    pointsPerGame DECIMAL(10,4),
    shotsPerGame DECIMAL(10,4),
    penaltyMinutesPerGame DECIMAL(10,4),
    hitsPerGame DECIMAL(10,4),
    blocksPerGame DECIMAL(10,4),
    timeOnIcePerGame DECIMAL(10,4),

    lastName VARCHAR(100),

    INDEX idx_season (season),
    INDEX idx_seasonId (seasonId),
    INDEX idx_gameTypeId (gameTypeId),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_positionCode (positionCode),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_skaterFullName (skaterFullName),

    INDEX idx_season_gameTypeId (season, gameTypeId),
    INDEX idx_season_gameDate (season, gameDate),
    INDEX idx_season_teamAbbrev (season, teamAbbrev),
    INDEX idx_season_playerId (season, playerId)
);

DELETE FROM wp_tsa_skater_scoringpergame
WHERE skaterFullName = 'skaterFullName';
