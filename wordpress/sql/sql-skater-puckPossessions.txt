DROP TABLE IF EXISTS wp_tsa_skater_puckpossessions;

CREATE TABLE wp_tsa_skater_puckpossessions (
    gameDate DATE,
    gameId BIGINT,
    skaterFullName VARCHAR(150),
    playerId BIGINT,
    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),
    shootsCatches VARCHAR(10),
    positionCode VARCHAR(10),
    gamesPlayed INT,

    timeOnIcePerGame5v5 DECIMAL(10,4),
    satPct DECIMAL(10,4),
    usatPct DECIMAL(10,4),
    goalsPct DECIMAL(10,4),
    individualSatForPer60 DECIMAL(10,4),
    individualShotsForPer60 DECIMAL(10,4),
    onIceShootingPct DECIMAL(10,4),
    zoneStartPct DECIMAL(10,4),
    offensiveZoneStartRatio DECIMAL(10,4),
    offensiveZoneStartPct DECIMAL(10,4),
    neutralZoneStartPct DECIMAL(10,4),
    defensiveZoneStartPct DECIMAL(10,4),
    faceoffPct5v5 DECIMAL(10,4),

    lastName VARCHAR(100),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_positionCode (positionCode),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_skaterFullName (skaterFullName)
);

DELETE FROM wp_tsa_skater_puckpossessions
WHERE skaterFullName = 'skaterFullName';
