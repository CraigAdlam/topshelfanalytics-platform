DROP TABLE IF EXISTS wp_tsa_skater_percentages;

CREATE TABLE wp_tsa_skater_percentages (
    season VARCHAR(7),
    seasonId INT,
    gameTypeId INT,

    gameDate DATE,
    gameId BIGINT,

    skaterFullName VARCHAR(150),
    playerId BIGINT,

    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),

    shootsCatches VARCHAR(10),
    positionCode VARCHAR(10),

    gamesPlayed INT,

    timeOnIcePerGame5v5 DECIMAL(10,4),

    satPercentage DECIMAL(10,4),
    satPercentageAhead DECIMAL(10,4),
    satPercentageTied DECIMAL(10,4),
    satPercentageBehind DECIMAL(10,4),
    satPercentageClose DECIMAL(10,4),
    satRelative DECIMAL(10,4),

    usatPercentage DECIMAL(10,4),
    usatPercentageAhead DECIMAL(10,4),
    usatPercentageTied DECIMAL(10,4),
    usatPercentageBehind DECIMAL(10,4),
    usatPrecentageClose DECIMAL(10,4),
    usatRelative DECIMAL(10,4),

    zoneStartPct5v5 DECIMAL(10,4),
    shootingPct5v5 DECIMAL(10,4),
    skaterSavePct5v5 DECIMAL(10,4),
    skaterShootingPlusSavePct5v5 DECIMAL(10,4),

    lastName VARCHAR(100),

    INDEX idx_season (season),
    INDEX idx_seasonId (seasonId),
    INDEX idx_gameTypeId (gameTypeId),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_positionCode (positionCode),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_skaterFullName (skaterFullName),

    INDEX idx_season_gameTypeId (season, gameTypeId),
    INDEX idx_season_gameDate (season, gameDate),
    INDEX idx_season_teamAbbrev (season, teamAbbrev),
    INDEX idx_season_playerId (season, playerId)
);

DELETE FROM wp_tsa_skater_percentages
WHERE skaterFullName = 'skaterFullName';
