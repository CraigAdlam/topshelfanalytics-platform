DROP TABLE IF EXISTS wp_tsa_team_shootout;

CREATE TABLE wp_tsa_team_shootout (
    gameDate DATE,
    gameId BIGINT,
    homeRoad VARCHAR(10),
    teamFullName VARCHAR(150),
    teamId BIGINT,
    opponentTeamAbbrev VARCHAR(10),
    gamesPlayed INT,
    wins INT,
    losses INT,
    otLosses INT,
    points INT,
    pointPct DECIMAL(10,4),
    shootoutGamesPlayed INT,
    shootoutWins INT,
    shootoutLosses INT,
    shootoutPoints INT,
    shootoutWinPct DECIMAL(10,4),
    shootoutGoals INT,
    shootoutShots INT,
    shootoutShootingPct DECIMAL(10,4),
    shootoutShotsAgainst INT,
    shootoutGoalsAgainst INT,
    shootoutSaves INT,
    shootoutSavePct DECIMAL(10,4),
    shootoutShootingPlusSavePct DECIMAL(10,4),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_teamId (teamId),
    INDEX idx_teamFullName (teamFullName),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad)
);

DELETE FROM wp_tsa_team_shootout
WHERE teamFullName = 'teamFullName';
