DROP TABLE IF EXISTS wp_tsa_team_powerplay;

CREATE TABLE wp_tsa_team_powerplay (
    gameDate DATE,
    gameId BIGINT,
    homeRoad VARCHAR(10),
    teamFullName VARCHAR(150),
    teamId BIGINT,
    opponentTeamAbbrev VARCHAR(10),
    gamesPlayed INT,
    wins INT,
    losses INT,
    otLosses INT,
    points INT,
    pointPct DECIMAL(10,4),
    ppOpportunities INT,
    powerPlayGoalsFor INT,
    shGoalsAgainst INT,
    ppNetGoals INT,
    ppTimeOnIcePerGame DECIMAL(10,4),
    ppOpportunitiesPerGame DECIMAL(10,4),
    ppGoalsPerGame DECIMAL(10,4),
    shGoalsAgainstPerGame DECIMAL(10,4),
    ppNetGoalsPerGame DECIMAL(10,4),
    powerPlayPct DECIMAL(10,4),
    powerPlayNetPct DECIMAL(10,4),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_teamId (teamId),
    INDEX idx_teamFullName (teamFullName),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad)
);

DELETE FROM wp_tsa_team_powerplay
WHERE teamFullName = 'teamFullName';
