DROP TABLE IF EXISTS wp_tsa_team_outshootoutshotby;

CREATE TABLE wp_tsa_team_outshootoutshotby (
    gameDate DATE,
    gameId BIGINT,
    homeRoad VARCHAR(10),
    teamFullName VARCHAR(150),
    teamId BIGINT,
    opponentTeamAbbrev VARCHAR(10),
    gamesPlayed INT,
    wins INT,
    losses INT,
    otLosses INT,
    points INT,
    pointPct DECIMAL(10,4),
    shotsForPerGame DECIMAL(10,4),
    shotsAgainstPerGame DECIMAL(10,4),
    netShotsPerGame DECIMAL(10,4),
    winsOutshootOpponent INT,
    lossesOutshootOpponent INT,
    tiesOutshootOpponent INT,
    otLossesOutshootOpponent INT,
    winsOutshotByOpponent INT,
    lossesOutshotByOpponent INT,
    tiesOutshotByOpponent INT,
    otLossesOutshotByOpponent INT,
    winsEvenShots INT,
    lossesEvenShots INT,
    tiesEvenShots INT,
    otLossesEvenShots INT,

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_teamId (teamId),
    INDEX idx_teamFullName (teamFullName),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad)
);

DELETE FROM wp_tsa_team_outshootoutshotby
