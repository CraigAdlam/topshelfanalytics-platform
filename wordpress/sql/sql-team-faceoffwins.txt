DROP TABLE IF EXISTS wp_tsa_team_faceoffwins;

CREATE TABLE wp_tsa_team_faceoffwins (
    gameDate DATE,
    gameId BIGINT,
    homeRoad VARCHAR(10),
    teamFullName VARCHAR(150),
    teamId BIGINT,
    opponentTeamAbbrev VARCHAR(10),
    gamesPlayed INT,
    totalFaceoffs INT,
    faceoffsWon INT,
    faceoffsLost INT,
    faceoffWinPct DECIMAL(10,4),
    evFaceoffs INT,
    evFaceoffsWon INT,
    evFaceoffsLost INT,
    ppFaceoffs INT,
    ppFaceoffsWon INT,
    ppFaceoffsLost INT,
    shFaceoffs INT,
    shFaceoffsWon INT,
    shFaceoffsLost INT,
    offensiveZoneFaceoffs INT,
    offensiveZoneFaceoffWins INT,
    offensiveZoneFaceoffLosses INT,
    neutralZoneFaceoffs INT,
    neutralZoneFaceoffWins INT,
    neutralZoneFaceoffLosses INT,
    defensiveZoneFaceoffs INT,
    defensiveZoneFaceoffWins INT,
    defensiveZoneFaceoffLosses INT,

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_teamId (teamId),
    INDEX idx_teamFullName (teamFullName),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad)
);

DELETE FROM wp_tsa_team_faceoffwins
WHERE teamFullName = 'teamFullName';
