DROP TABLE IF EXISTS wp_tsa_top_picks_2plus;

CREATE TABLE wp_tsa_top_picks_2plus (
    predictionDate DATE,
    dataCutoffDate DATE,

    playerId BIGINT,
    skaterFullName VARCHAR(150),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),

    homeRoad VARCHAR(10),
    positionCode VARCHAR(10),

    predProb2Plus DECIMAL(10,6),

    accuracy2Plus DECIMAL(10,6),
    precision2Plus DECIMAL(10,6),
    recall2Plus DECIMAL(10,6),
    f1Score2Plus DECIMAL(10,6),

    gameId BIGINT,
    skaterFullName_actual VARCHAR(150),

    modelGames2Plus INT,

    actualShots INT,
    hit2Plus TINYINT,
    resultLabel VARCHAR(20),

    shotsMean DECIMAL(10,4),
    shotsMedian DECIMAL(10,4),
    shotsMode DECIMAL(10,4),
    shotsConcentration DECIMAL(10,6),
    shotsSkew DECIMAL(10,6),
    gamesInDistribution INT,

    predProb2PlusPct DECIMAL(10,4),
    accuracy2PlusPct DECIMAL(10,4),
    precision2PlusPct DECIMAL(10,4),
    recall2PlusPct DECIMAL(10,4),
    f1Score2PlusPct DECIMAL(10,4),
    shotsConcentrationPct DECIMAL(10,4),

    correctPick TINYINT,

    runTimestamp VARCHAR(50),
    modelVersion VARCHAR(100),
    canonicalThreshold DECIMAL(10,4),

    INDEX idx_predictionDate (predictionDate),
    INDEX idx_playerId (playerId),
    INDEX idx_skater (skaterFullName),
    INDEX idx_team (teamAbbrev),
    INDEX idx_result (resultLabel),

    INDEX idx_prob (predProb2Plus),
    INDEX idx_precision (precision2Plus),
    INDEX idx_f1 (f1Score2Plus),

    INDEX idx_date_prob (
        predictionDate,
        predProb2Plus
    )
);

DELETE FROM wp_tsa_top_picks_2plus
WHERE skaterFullName = 'skaterFullName';
