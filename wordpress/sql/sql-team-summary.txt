DROP TABLE IF EXISTS wp_tsa_team_summary;

CREATE TABLE wp_tsa_team_summary (
    season VARCHAR(7),
    seasonId INT,
    gameTypeId INT,

    gameDate DATE,
    gameId BIGINT,

    homeRoad VARCHAR(10),

    teamFullName VARCHAR(150),
    teamAbbrev VARCHAR(10),
    teamId BIGINT,

    opponentTeamAbbrev VARCHAR(10),

    gamesPlayed INT,
    wins INT,
    losses INT,
    otLosses INT,

    points INT,
    pointPct DECIMAL(10,4),

    winsInRegulation INT,
    regulationAndOtWins INT,
    winsInShootout INT,

    goalsFor INT,
    goalsAgainst INT,

    goalsForPerGame DECIMAL(10,4),
    goalsAgainstPerGame DECIMAL(10,4),

    teamShutouts INT,

    powerPlayPct DECIMAL(10,4),
    penaltyKillPct DECIMAL(10,4),
    powerPlayNetPct DECIMAL(10,4),
    penaltyKillNetPct DECIMAL(10,4),

    shotsForPerGame DECIMAL(10,4),
    shotsAgainstPerGame DECIMAL(10,4),

    faceoffWinPct DECIMAL(10,4),

    INDEX idx_season (season),
    INDEX idx_seasonId (seasonId),
    INDEX idx_gameTypeId (gameTypeId),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_teamId (teamId),
    INDEX idx_teamFullName (teamFullName),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad),

    INDEX idx_season_gameTypeId (season, gameTypeId),
    INDEX idx_season_gameDate (season, gameDate),
    INDEX idx_season_teamId (season, teamId)
);

DELETE FROM wp_tsa_team_summary
WHERE teamFullName = 'teamFullName';
