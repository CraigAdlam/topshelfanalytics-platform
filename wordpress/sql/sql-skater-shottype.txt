DROP TABLE IF EXISTS wp_tsa_skater_shottype;

CREATE TABLE wp_tsa_skater_shottype (
    season VARCHAR(7),
    seasonId INT,
    gameTypeId INT,

    gameDate DATE,
    gameId BIGINT,

    skaterFullName VARCHAR(150),
    playerId BIGINT,

    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),

    gamesPlayed INT,

    goals INT,
    goalsWrist INT,
    goalsSnap INT,
    goalsSlap INT,
    goalsBackhand INT,
    goalsTipIn INT,
    goalsDeflected INT,
    goalsWrapAround INT,
    goalsPoke INT,
    goalsCradle INT,
    goalsBetweenLegs INT,
    goalsBat INT,

    shotsOnNetWrist INT,
    shotsOnNetSnap INT,
    shotsOnNetSlap INT,
    shotsOnNetBackhand INT,
    shotsOnNetTipIn INT,
    shotsOnNetDeflected INT,
    shotsOnNetWrapAround INT,
    shotsOnNetPoke INT,
    shotsOnNetCradle INT,
    shotsOnNetBetweenLegs INT,
    shotsOnNetBat INT,

    shootingPct DECIMAL(10,4),
    shootingPctWrist DECIMAL(10,4),
    shootingPctSnap DECIMAL(10,4),
    shootingPctSlap DECIMAL(10,4),
    shootingPctBackhand DECIMAL(10,4),
    shootingPctTipIn DECIMAL(10,4),
    shootingPctDeflected DECIMAL(10,4),
    shootingPctWrapAround DECIMAL(10,4),
    shootingPctPoke DECIMAL(10,4),
    shootingPctCradle DECIMAL(10,4),
    shootingPctBetweenLegs DECIMAL(10,4),
    shootingPctBat DECIMAL(10,4),

    lastName VARCHAR(100),

    INDEX idx_season (season),
    INDEX idx_seasonId (seasonId),
    INDEX idx_gameTypeId (gameTypeId),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_skaterFullName (skaterFullName),

    INDEX idx_season_gameTypeId (season, gameTypeId),
    INDEX idx_season_gameDate (season, gameDate),
    INDEX idx_season_teamAbbrev (season, teamAbbrev),
    INDEX idx_season_playerId (season, playerId)
);

DELETE FROM wp_tsa_skater_shottype
WHERE skaterFullName = 'skaterFullName';
