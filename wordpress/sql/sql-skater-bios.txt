DROP TABLE IF EXISTS wp_tsa_skater_bios;

CREATE TABLE wp_tsa_skater_bios (
    season VARCHAR(7),
    seasonId INT,
    gameTypeId INT,

    skaterFullName VARCHAR(150),
    playerId BIGINT,

    currentTeamAbbrev VARCHAR(10),
    currentTeamName VARCHAR(100),

    shootsCatches VARCHAR(10),
    positionCode VARCHAR(10),

    birthDate DATE,
    birthCity VARCHAR(100),
    birthStateProvinceCode VARCHAR(20),
    birthCountryCode VARCHAR(10),
    nationalityCode VARCHAR(10),

    height VARCHAR(20),
    weight INT,

    draftYear INT,
    draftRound INT,
    draftOverall INT,

    firstSeasonForGameType VARCHAR(10),
    isInHallOfFameYn VARCHAR(5),

    gamesPlayed INT,
    goals INT,
    assists INT,
    points INT,

    lastName VARCHAR(100),

    INDEX idx_season (season),
    INDEX idx_seasonId (seasonId),
    INDEX idx_gameTypeId (gameTypeId),

    INDEX idx_playerId (playerId),
    INDEX idx_skaterFullName (skaterFullName),
    INDEX idx_currentTeamAbbrev (currentTeamAbbrev),
    INDEX idx_positionCode (positionCode),
    INDEX idx_lastName (lastName),

    INDEX idx_season_gameTypeId (season, gameTypeId),
    INDEX idx_season_playerId (season, playerId),
    INDEX idx_season_teamAbbrev (season, currentTeamAbbrev)
);

DELETE FROM wp_tsa_skater_bios
WHERE skaterFullName = 'skaterFullName';
