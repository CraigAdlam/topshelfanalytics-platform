DROP TABLE IF EXISTS wp_tsa_team_scoretrailfirst;

CREATE TABLE wp_tsa_team_scoretrailfirst (
    season VARCHAR(7),
    seasonId INT,
    gameTypeId INT,

    gameDate DATE,
    gameId BIGINT,

    homeRoad VARCHAR(10),

    teamFullName VARCHAR(150),
    teamAbbrev VARCHAR(10),
    teamId BIGINT,

    opponentTeamAbbrev VARCHAR(10),

    gamesPlayed INT,

    wins INT,
    losses INT,
    otLosses INT,
    points INT,

    scoringFirstGamesPlayed INT,

    winsScoringFirst INT,
    lossesScoringFirst INT,
    tiesScoringFirst INT,
    otLossesScoringFirst INT,

    winPctScoringFirst DECIMAL(10,4),

    trailingFirstGamesPlayed INT,

    winsTrailingFirst INT,
    lossesTrailingFirst INT,
    tiesTrailingFirst INT,
    otLossesTrailingFirst INT,

    winPctTrailingFirst DECIMAL(10,4),

    INDEX idx_season (season),
    INDEX idx_seasonId (seasonId),
    INDEX idx_gameTypeId (gameTypeId),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),

    INDEX idx_teamId (teamId),
    INDEX idx_teamFullName (teamFullName),
    INDEX idx_teamAbbrev (teamAbbrev),

    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad),

    INDEX idx_season_gameTypeId (season, gameTypeId),
    INDEX idx_season_gameDate (season, gameDate),
    INDEX idx_season_teamId (season, teamId),
    INDEX idx_season_teamAbbrev (season, teamAbbrev)
);

DELETE FROM wp_tsa_team_scoretrailfirst
WHERE teamFullName = 'teamFullName';
